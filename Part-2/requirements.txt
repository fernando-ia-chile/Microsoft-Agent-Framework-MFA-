# pero en un entorno limpio new_14 fallaría con ImportError sin esta línea.
opentelemetry-sdk==1.44.0

# Model Context Protocol — necesario para new_15 (MCPStdioTool del lado cliente y
# FastMCP en mcp_calculator_server.py). Mismo caso que el SDK de OpenTelemetry:

# Utilidades
python-dotenv>=1.0.0   # Soporte para archivos .env
